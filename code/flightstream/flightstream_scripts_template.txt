# SCRIPT: FlightStream Canard Configuration - Mesh Preprocessing Template
# AUTHOR: Omar Hernando de la Torre
# DATE: April 2026
# DESCRIPTION: Imports the component CCS meshes of the canard aircraft,
#              applies the initial surface cleaning operations and prepares
#              the model for manual symmetry inspection before solver setup.

SURFACE_CLEARALL

CCS_IMPORT
CLOSE_COMPONENT_ENDS ENABLE
UPDATE_PROPERTIES ENABLE
CLEAR_EXISTING DISABLE
FILE <CANARD_AIRCRAFT_FUSELAGE_CCS_RUTE_FILE>

CCS_IMPORT
CLOSE_COMPONENT_ENDS ENABLE
UPDATE_PROPERTIES ENABLE
CLEAR_EXISTING DISABLE
FILE <CANARD_AIRCRAFT_WING_CCS_RUTE_FILE>

CCS_IMPORT
CLOSE_COMPONENT_ENDS ENABLE
UPDATE_PROPERTIES ENABLE
CLEAR_EXISTING DISABLE
FILE <CANARD_AIRCRAFT_CANARD_CCS_RUTE_FILE>

CCS_IMPORT
CLOSE_COMPONENT_ENDS ENABLE
UPDATE_PROPERTIES ENABLE
CLEAR_EXISTING DISABLE
FILE <CANARD_AIRCRAFT_VTP_CCS_RUTE_FILE>

SURFACE_INVERT -1

SURFACE_RENAME 1 Fuselage
SURFACE_RENAME 2 Wing
SURFACE_RENAME 3 Canard
SURFACE_RENAME 4 Vtp

BOOLEAN_UNITE_MESH -1
DELETE_DEGENERATE_FACES
SELECT_GEOMETRY_BY_ID -1

SURFACE_CUT_BY_PLANE
FRAME 1
PLANE XZ
OFFSET 0.0
SURFACE -1


# ==========================================================================
# SCRIPT: FlightStream Canard Configuration - Solver Setup and Sweep Template
# AUTHOR: Omar Hernando de la Torre
# DATE: April 2026
# DESCRIPTION: Defines the aerodynamic boundary conditions, cruise fluid
#              properties, reference quantities, centre of gravity frame and
#              angle-of-attack sweep for the canard aircraft.
# ==========================================================================

AUTO_DETECT_TRAILING_EDGES
AUTO_DETECT_WAKE_TERMINATION_NODES

FLUID_PROPERTIES
DENSITY <DENSITY_KG_M3>
PRESSURE <PRESSURE_PA>
TEMPERATURE <TEMPERATURE_K>
VISCOSITY <DYNAMIC_VISCOSITY_KG_MS>
SPECIFIC_HEAT_RATIO <SPECIFIC_HEAT_RATIO>

CREATE_NEW_COORDINATE_SYSTEM
SET_COORDINATE_SYSTEM_NAME 2 <CG_FRAME_NAME>

EDIT_COORDINATE_SYSTEM
FRAME 2
NAME <CG_FRAME_NAME>
ORIGIN_X <CG_X_CANARD>
ORIGIN_Y <CG_Y_CANARD>
ORIGIN_Z <CG_Z_CANARD>
VECTOR_X_X 1
VECTOR_X_Y 0
VECTOR_X_Z 0
VECTOR_Y_X 0
VECTOR_Y_Y 1
VECTOR_Y_Z 0
VECTOR_Z_X 0
VECTOR_Z_Y 0
VECTOR_Z_Z 1

SOLVER_SET_REF_LENGTH <REFERENCE_LENGTH>
SOLVER_SET_REF_AREA <REFERENCE_AREA>
SET_SOLVER_ANALYSIS_LOADS_FRAME 2
SET_VORTICITY_DRAG_BOUNDARIES <NUMBER_OF_VORTICITY_DRAG_BOUNDARIES>
<VORTICITY_DRAG_BOUNDARY_IDS>
SET_BOUNDARY_LAYER_TYPE TURBULENT
SET_SOLVER_VISCOUS_COUPLING ENABLE
SET_AXIAL_SEPARATION_BOUNDARIES <NUMBER_OF_AXIAL_SEPARATION_BOUNDARIES>
<AXIAL_SEPARATION_BOUNDARY_IDS>
SET_CROSSFLOW_SEPARATION_DIAMETER <CROSSFLOW_SEPARATION_DIAMETER>
SET_CROSSFLOW_SEPARATION_BOUNDARIES <NUMBER_OF_CROSSFLOW_SEPARATION_BOUNDARIES>
<CROSSFLOW_SEPARATION_BOUNDARY_IDS>

INITIALIZE_SOLVER
SOLVER_MODEL SUBSONIC_PRANDTL_GLAUERT
SURFACES -1
WAKE_TERMINATION_X DEFAULT
SYMMETRY MIRROR
WALL_COLLISION_AVOIDANCE ENABLE
STABILIZATION ENABLE 1.000

EXECUTE_SOLVER_SWEEPER
ANGLE_OF_ATTACK ENABLE
SIDE_SLIP_ANGLE DISABLE
VELOCITY ENABLE
ANGLE_OF_ATTACK_START <AOA_START>
ANGLE_OF_ATTACK_STOP <AOA_STOP>
ANGLE_OF_ATTACK_DELTA <AOA_DELTA>
SIDE_SLIP_ANGLE_START <BETA_START>
SIDE_SLIP_ANGLE_STOP <BETA_STOP>
SIDE_SLIP_ANGLE_DELTA <BETA_DELTA>
VELOCITY_START <VELOCITY_START>
VELOCITY_STOP <VELOCITY_STOP>
VELOCITY_DELTA <VELOCITY_DELTA>
EXPORT_SURFACE_DATA_PER_STEP DISABLE
CLEAR_SOLUTION_AFTER_EACH_RUN DISABLE
REFERENCE_VELOCITY_EQUALS_FREESTREAM ENABLE
APPEND_TO_EXISTING_SWEEP DISABLE
<CANARD_AIRCRAFT_OUTPUT_CSV>
